# list和str相互转换
```
a = ['1', '2', '3', '', '5']
b = ','.join(a)
print(b, type(b), len(b))
c = b.split(',')
print(c, type(c), len(c))

d = ['6']
e = ','.join(d)
print(e)
```
```
>>1,2,3,,5 <class 'str'> 8
>>['1', '2', '3', '', '5'] <class 'list'> 5
>>6
```
